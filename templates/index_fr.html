function calculFc28() { const titreProjet = document.getElementById('titre_projet').value || "Non renseigné"; const lieuTest = document.getElementById('lieu_test').value || "Non renseigné"; const dateCoulageStr = document.getElementById('date_coulage').value; const dateTestStr = document.getElementById('date_test').value; const resJour = parseFloat(document.getElementById('resistance').value); const fc28Projet = parseFloat(document.getElementById('fc28').value); if(!dateCoulageStr || !dateTestStr || isNaN(resJour) || isNaN(fc28Projet) || resJour<=0 || fc28Projet<=0){ document.getElementById('result').innerText = "⚠️ Veuillez entrer des valeurs valides."; document.getElementById('result').style.color = "red"; return; } const dateCoulage = new Date(dateCoulageStr); const dateTest = new Date(dateTestStr); // Calcul automatique de l'âge du béton const age = Math.round((dateTest - dateCoulage)/(1000*60*60*24)); if(age <= 0){ document.getElementById('result').innerText = "⚠️ La date de test doit être après la date de coulage."; document.getElementById('result').style.color = "red"; return; } // Calcul exact Fc28 const res28 = resJour * Math.pow(28/age, 0.5); const conforme = res28 >= fc28Projet; const resultDiv = document.getElementById('result'); resultDiv.innerHTML = ` Projet : ${titreProjet} Endroit du test : ${lieuTest} Date de coulage : ${dateCoulage.toLocaleDateString('fr-FR')} Date du test : ${dateTest.toLocaleDateString('fr-FR')} Âge du béton : ${age} jours Fc estimée : ${res28.toFixed(2)} MPa Fc28 projet : ${fc28Projet} MPa ${conforme ? 'Résultat : Conforme' : 'Résultat : Non Conforme'} `; lastResult = { titreProjet, lieuTest, dateCoulage: dateCoulage.toLocaleDateString('fr-FR'), dateTest: dateTest.toLocaleDateString('fr-FR'), age, res28: res28.toFixed(2), fc28Projet, conforme }; } function exportPDF(){ if(!lastResult) { alert("Veuillez calculer le résultat avant d'exporter."); return; } const { jsPDF } = window.jspdf; const doc = new jsPDF(); // Logo SOFAZ en haut const img = document.getElementById("logo_header"); const canvas = document.createElement("canvas"); canvas.width = img.width; canvas.height = img.height; const ctx = canvas.getContext("2d"); ctx.drawImage(img,0,0,img.width,img.height); const imgData = canvas.toDataURL("image/png"); doc.addImage(imgData, "PNG", 15, 10, 50, 25); doc.setFontSize(16); doc.text("Rapport - SOFAZ Fc28 Tracker (TM)", 105, 50, null, null, "center"); doc.text("Développé par Ing. Fathi ZERRIOUH", 105, 60, null, null, "center"); doc.text("---------------------------------------------", 20, 70); doc.setFontSize(14); const startY = 80; const lineHeight = 10; doc.text(`Projet : ${lastResult.titreProjet}`, 20, startY); doc.text(`Endroit du test : ${lastResult.lieuTest}`, 20, startY + lineHeight); doc.text(`Date de coulage : ${lastResult.dateCoulage}`, 20, startY + lineHeight*2); doc.text(`Date du test : ${lastResult.dateTest}`, 20, startY + lineHeight*3); doc.text(`Âge du béton : ${lastResult.age} jours`, 20, startY + lineHeight*4); doc.text(`Fc estimée : ${lastResult.res28} MPa`, 20, startY + lineHeight*5); doc.text(`Fc28 projet : ${lastResult.fc28Projet} MPa`, 20, startY + lineHeight*6); doc.setTextColor(lastResult.conforme ? 0 : 255, 0, 0); // rouge si non conforme doc.text(lastResult.conforme ? "Résultat : Conforme" : "Résultat : Non Conforme", 20, startY + lineHeight*7); doc.save("rapport_fc28.pdf"); }
